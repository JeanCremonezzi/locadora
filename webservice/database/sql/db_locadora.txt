CREATE DATABASE locadora;

USE locadora;

CREATE TABLE pessoa (
	id INT AUTO_INCREMENT,
	nome VARCHAR(255),
	login VARCHAR(255),
	senha VARCHAR(255),
CONSTRAINT pessoa_pk PRIMARY KEY(id)
);

CREATE TABLE carro (
	id INT AUTO_INCREMENT,
	nome VARCHAR(255),
	marca VARCHAR(255),
	ano YEAR,
	pessoa INT,
CONSTRAINT carro_pk PRIMARY KEY(id),
CONSTRAINT pessoa_fk FOREIGN KEY (pessoa) REFERENCES pessoa(id)
);
